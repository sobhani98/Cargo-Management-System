Create table agency (agency_id int (11) not null, agency_name varchar (30) not null, contact_number int (11) not null, address varchar (40) not null, primary key (agency_id));
create table F_harbour (fh_id int (5) not null, fh_name varchar (20) not null, fh_loc varchar (30) not null, agency_id int not null, primary key (fh_id), foreign key (agency_id) references agency (agency_id) on delete cascade);
create table T_harbour (th_id int (5) not null, th_name varchar (20) not null, th_loc varchar (30) not null, agency_id int not null, primary key (th_id), foreign key(agency_id) references agency  (agency_id) on delete cascade);
create table ship_owner (so_id int (5) not null, so_naame varchar (20) not null, so_address varchar (30) not null, primary key (so_id));
create table ship_owner (so_id int (5) not null, so_naame varchar (20) not null, so_address varchar (30) not null, primary key (so_id));
create table ship (s_id int (5) not null, s_name varchar (20) not null, s_cap int (11) not null, fh_id int (5) not null, th_id int (5) not null, so_id int (5) not null, primary key (s_id), foreign key (fh_id) references F_harbour (fh_id) on delete cascade, foreign key (th_id) references T_harbour (th_id) on delete cascade, foreign key (so_id) references ship_owner (so_id) on delete cascade);
create table customer (cust_id int (5) not null, cust_name varchar (20) not null, cust_add varchar (30) not null, primary key (cust_id));
create table cargo (c_id int (5) not null, c_name varchar (20) not null, c_price int (11) not null, c_weight int (5) not null, c_status varchar (30) not null, s_id int (5) not null, cust_id int (5) not null, primary key (c_id), foreign key (s_id) references ship (s_id) on delete cascade, foreign key (cust_id) references customer (cust_id) on delete cascade);
create table billing (bill_no int (11) not null, bill_date varchar (20) not null, shipping_charge int (10) not null, c_id int (5) not null, primary key (bill_no), foreign key (c_id) references cargo (c_id));
